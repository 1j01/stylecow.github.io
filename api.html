.is()

Check whether the element matches with the provided conditions. The conditions are defined in an object with the following properties:

  • type: to test the element type (Rule, Declaration, Function, etc)
  • name: to test the element name (some elements like Rule or Selector don't have name)
  • vendor: to test the element vendor prefix
  • string: to test the stringified element

Usage example:

body {
    border-radius: 3px;
    font-size: 1em;
    border-color: rgba(2, 34, 255, 0.5);
    border-radius: 4px;
}   

//Get the first declaration:
var declaration = code.searchFirst('Declaration');

//Use an object to check conditions:
console.log(declaration.is({
    type: 'Declaration',
    name: 'border-radius'
}));
// true

//You can use a string to check only the type
console.log(declaration.is('Declaration'));
// true

//Check the stringify object
console.log(declaration.is({
    string: 'border-radius: 3px;'
}));
// true

//Check the stringify object
console.log(declaration.is({
    string: 'border-radius: 4px;'
}));
// false

//Use arrays to search between various values:
console.log(declaration.is({
    name: ['border-radius', 'font-size']
}));
// true

//Or even a regular expression:
console.log(declaration.is({
    name: /^border-/
}));
// true

.children()

Returns a collection with the direct children of an object. You can provide a filter to return only the children matching with some conditions.

body {
    color: blue;
}

@media (min-width: 200px) {
    body {
        color: red;
    }
}
@font-face {
    font-family: 'myFont';
    src: url(myfont.woff);
}

//Get all children:
var result = code.children();

console.log(result.length); //3

//Get only the objects of type "AtRule"
var result = code.children('AtRule');

console.log(result.length); //2

//Get only the @media objects:
var result = code.children({
    type: 'AtRule',
    name: 'media'
});

console.log(result.length); //1

.firstChild()

It's like .children() but returns the first result found instead an a collection with all elements, and of course, you can provide a filter.

.hasChild()

Returns true if the object has at least one child matching with the filter. For example:

body {
    border-radius: 3px;
    font-size: 1em;
}   

var ruleBlock = code.firstChild('Block');

//Check if the rule block has a "border-radius" statement
ruleBlock.hasChild({type: 'Declaration', name: 'border-radius'}); //true

//Check if the rule has a "color" statement
ruleBlock.hasChild({type: 'Declaration', name: 'color'}); //false

.search()

It's like .children() but search recursively in cascade and returns a collection with all results found.

body {
    color: blue;
}
p {
    border-radius: 3px solid blue;
    background: linear-gradient(blue, red);
}

//Search all keywords named "blue" and change to "green":
code.search({
    type: 'Keyword',
    name: 'blue'
}).forEach(function (keyword) {
    keyword.name = 'green';
});

//Get the result code:
console.log(code.toString());
/*
body {
    color: green;
}
p {
    border-radius: 3px solid green;
    background: linear-gradient(green, red);
}
*/

.searchFirst()

It's like .search() but returns the first element found.

.has()

Returns true if the object has at least one element matching with the filter. It's like .hasChild() but search recursively in all descendents

.prev() | .next()

Returns the inmediate following or preceding sibling of the element. For example:

body {
    color: blue;
    font-family: Arial;
    border: solid 1px red;
}

//Search the font-family declaration:
var declaration = code.searchFirst({
    type: 'Declaration',
    name: 'font-family'
});

//Get the next sibling
var next = declaration.next();

console.log(next.toString()); // "border: solid 1px red"

//Get the previous sibling
var prev = declaration.prev();

console.log(prev.toString()); // "color: blue"

.parent()

Returns the parent element. This function accepts a filter to return the first parent matching. For example:

body {
    border: solid 1px red;
}

//Search the "red" keyword:
var red = code.searchFirst({
    type: 'Keyword',
    name: 'red'
});

//Get the inmediate parent:
console.log(red.parent().type); // Value

//Get the declaration parent:
console.log(red.parent('Declaration').type); // Declaration

.clone()

Creates a deep clone of the object (with all children)

body {
    border-radius: 1px;
}

//Get the border-radius declaration:
var declaration = code.searchFirst({type: 'Declaration', name: 'border-radius'});

//Clone the declaration
var cloned = declaration.clone();
cloned.name = '-moz-border-radius';

console.log(declaration.toString()); // border-radius: 1px;
console.log(cloned.toString()); // -moz-border-radius: 1px;

.cloneBefore() | .cloneAfter()

Creates a deep clone of the object and insert as a sibling after or before the object.

body {
    border-radius: 1px;
}

//Get the border-radius declaration:
var declaration = code.searchFirst({type: 'Declaration', name: 'border-radius'});

//Clone the declaration, insert before and change the name:
declaration.cloneBefore().name = '-moz-border-radius';

console.log(code.toString());

/*
body {
    -moz-border-radius: 1px;
    border-radius: 1px;
}
*/

//Clone the declaration, insert after and change the name:
declaration.cloneAfter().name = '-webkit-border-radius';

console.log(code.toString());

/*
body {
    -moz-border-radius: 1px;
    border-radius: 1px;
    -webkit-border-radius: 1px;
}
*/

.push() | .unshift() | .splice()

This functions are like the Array.prototype ones, but only css elements object are allowed.

//Creates a new declaration
var declaration = stylecow.Declaration.createFromString('font-family: Arial');

//Creates a new value
var value = new stylecow.Value.createFromString('sans-serif');

//Push the value into the declaration
declaration.push(value);

console.log(declaration.toString()); // font-family: Arial, sans-serif;

//Creates a new value
var value2 = new stylecow.Value.createFromString('Helvetica');

//Unshift the value into the declaration
declaration.unshift(value2);

console.log(declaration.toString()); // font-family: Helvetica, Arial, sans-serif;

.before() | .after()

Adds new children as siblings after or before the object.

body {
    border-radius: 1px;
}

//Get the border-radius declaration:
var declaration = code.searchFirst({type: 'Declaration', name: 'border-radius'});

//Insert a new element before:
declaration.before(stylecow.Declaration.createFromString('color: blue'));

console.log(code.toString());
/*
body {
    color: blue;
    border-radius: 1px;
}
*/

//Insert a new element after
declaration.after(stylecow.Declaration.createFromString('background: gray'));

console.log(code.toString());
/*
body {
    color: blue:
    border-radius: 1px;
    background: gray;
}
*/

.replaceWith()

Replaces the current object with a new one.

body {
    font-size: 1em;
    color: blue;
}

//Get the color declaration:
var declaration = code.searchFirst({type: 'Declaration', name: 'color'});

//Replace the declaration by other:
declaration.replaceWith(stylecow.Declaration.createFromString('background: red'));

console.log(code.toString());
/*
body {
    font-size: 1em;
    background: red;
}
*/

.remove()

Removes the element.

body {
    font-size: 1em;
    color: blue;
}

//Get the color declaration:
var declaration = code.searchFirst({type: 'Declaration', name: 'color'});

//Remove it
declaration.remove();

console.log(code.toString());
/*
body {
    font-size: 1em;
}
*/

.detach()

Detach the element from its parent. It's similar to .remove() but it less destructive (.remove() destroy the object and it's children recursively).

.empty()

Removes all children of the element.

body {
    font-size: 1em;
    color: blue;
}

//Empty the rule
code.searchFirst({type: 'Rule'}).empty();

console.log(code.toString());
/*
body {
}
*/

.toArray()

Returns an array with all children of the element as a string.

body {
    background: rgba(100, 150, 200, 0.5);
}

//Get the rgba function:
var rgba = code.searchFirst({type: 'Function', name: 'rgba'});

console.log(rgba.toArray());
/*
['100', '150', '200', '0.5']
*/

.setData() | .getData()

Saves arbitrary data in the element. This data can be recovered in the cascade tree for any of its children.

The function .getData() returns any value saved. If the element has not the value, it search the value in its parents.

body {
    color: blue;

    p {
        font-family: Arial;

        strong {
            font-weight: bold;
        }
    }
}

//Save a value in the root
code.setData('bar', 'foo');

//Select the font-weight declaration:
var font_weight = code.searchFirst({type: 'Declaration', name: 'font-weight'});

console.log(font_weight.getData('bar')); //foo

.index()

Returns the (zero based) index position of the element in its parent. If the element has not parent, returns -1

body {
    color: blue;
    background: red;
}

console.log(code.searchFirst({type: 'Declaration', name: 'color'}).index());
// 0

console.log(code.searchFirst({type: 'Declaration', name: 'background'}).index());
// 1

.set()

Set properties to the element. Useful to modifications in a chain:

body {
    border-radius: 1px;
}

//Get the border-radius declaration:
var declaration = code.searchFirst({type: 'Declaration', name: 'border-radius'});

//Clone before and modify the name
declaration.cloneBefore().set('name', '-moz-border-radius');

console.log(code.toString());
/*
body {
    -moz-border-radius: 1px;
    border-radius: 1px;
}
*/
